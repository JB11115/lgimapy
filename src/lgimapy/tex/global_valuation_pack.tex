%% Define function for perctile bars in tables.
\newlength{\barw}
\setlength{\barw}{0.15mm}
\def\pctbar#1{%%
	{\color{gray}\rule{#1\barw}{10pt}} #1\%}

%% Define function for divergent bars in tables.
\def\bar#1#2{%
	\color{#2}\rule{#1\barw}{8pt}
}

\def\divbar#1#2{%
	\ifnum#1<0{%
		\bar{\the\numexpr50+#1}{white}
		\bar{-#1}{#2}
		\bar{50}{white}
	}
	\else{%
		\bar{50}{white}
		\bar{#1}{#2}
		\bar{\the\numexpr50-#1}{white}
	}
	\fi
}
